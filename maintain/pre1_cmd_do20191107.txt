
#2019-11-07 向集控转发添加防火墙规则
#--------------------------------------------------------------------------------
#remote ip: 198.1.3.5     198.1.3.6  gateway:198.1.3.254  netmask:255.255.255.0
#local: 192.168.1.85   gateway:192.168.1.254  netmask:255.255.255.0
#ifconfig: eth6
#
#protocl: sftp
#data director: zfmd put file to remote
#
#--------------------------------------------------------------------------------
#--------------------------------------------------------------------------------
#--------------------------------------------------------------------------------
#--------------------------------------------------------------------------------
#--------------------------------------------------------------------------------
#



iptables -A OUTPUT -d 198.1.3.5,198.1.3.6 -s 192.168.1.85 -o eth6 -p tcp  --dport 22  -m state --state NEW,ESTABLISHED -j ACCEPT
iptables -A INPUT -s 198.1.3.5,198.1.3.6 -d 192.168.1.85 -i eth6 -p tcp  --sport 22  -m state --state ESTABLISHED -j ACCEPT
/etc/rc.d/init.d/iptables save
service iptables restart
iptables -L -n -v
