select tablespace_name, 
       file_name, 
       bytes / 1024 / 1024 || 'M' as MB
  from dba_data_files;
  
  
--增加某个表空间的数据文件（下面语句表空间是USERS）
ALTER TABLESPACE USERS ADD DATAFILE
    '/zfmd/wpfs20/oradata/mqzorcl/users02.dbf'
  size 7167M autoextend on
